  updateAnalytics();

  // Set current season highlight
  const month = new Date().getMonth();
  const seasons = document.querySelectorAll('.season-item');
  const seasonIdx = month < 3 ? 3 : month < 6 ? 0 : month < 9 ? 1 : 2;
  seasons.forEach(s => s.classList.remove('active'));
  seasons[seasonIdx]?.classList.add('active');
});

// ── Navigation ────────────────────────────────────────────────
function showPage(id) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
  document.getElementById('page-' + id).classList.add('active');
  document.getElementById('nav-' + id).classList.add('active');
  if (id === 'analytics') updateAnalytics();
  if (id === 'history')   renderHistory();
}

// ── Health check ──────────────────────────────────────────────
async function checkHealth() {
  try {
    const d = await fetch('/api/health').then(r => r.json());
    const el = document.getElementById('modelStatus');
    if (d.model_loaded) {
      el.textContent = `Model Ready · ${d.num_classes} Classes`;
    } else {
      el.textContent = 'Demo Mode — No Model';
    }
  } catch(e) {
    document.getElementById('modelStatus').textContent = 'Server Offline';
  }
}

  document.getElementById('dropZone').classList.remove('drag-over');
});
document.getElementById('dropZone').addEventListener('drop', e => {
  e.preventDefault();
  document.getElementById('dropZone').classList.remove('drag-over');
  const f = e.dataTransfer.files[0];
  if (f && f.type.startsWith('image/')) handleFile(f);

function handleFile(file) {
  state.selectedFile = file;
  const reader = new FileReader();
  reader.onload = e => {
    document.getElementById('previewImg').src = e.target.result;
    const img = new Image();
    img.onload = () => document.getElementById('mDims').textContent = `${img.width}×${img.height}px`;
    img.src = e.target.result;
  };
  reader.readAsDataURL(file);
  document.getElementById('mName').textContent = file.name.length > 18 ? file.name.slice(0,16)+'…' : file.name;
  document.getElementById('mSize').textContent = formatBytes(file.size);
  document.getElementById('mType').textContent = file.type || 'image/*';
  showZone('preview');
  document.getElementById('uploadSubtitle').textContent = 'Image ready';
}

function formatBytes(b) {
  if (b < 1024) return b + ' B';
  if (b < 1048576) return (b/1024).toFixed(1) + ' KB';
  return (b/1048576).toFixed(1) + ' MB';
}

// ── Analyze ───────────────────────────────────────────────────
async function analyzeImage() {
  if (!state.selectedFile) return;
  showZone('loading');
  animateLoadingBar();

  try {
    const form = new FormData();
    form.append('image', state.selectedFile);
    const res  = await fetch('/api/predict', { method:'POST', body: form });
    const data = await res.json();
    if (data.success) {
      state.currentResult = data.result;
      saveToHistory(data.result, document.getElementById('previewImg').src);
      updateStats();
      setTimeout(() => renderResult(data.result), 500);
    } else {
      showZone('preview');
      toast('error', '❌ ' + data.error);
    }
  } catch(err) {
    showZone('preview');
    toast('error', '❌ Connection error: ' + err.message);
  }
}

function animateLoadingBar() {
  const bar = document.getElementById('loadBar');
  const steps = ['ls1','ls2','ls3','ls4'];
  let p = 0;
  const iv = setInterval(() => {
    p = Math.min(p + 15, 90);
    bar.style.width = p + '%';
  }, 200);
  steps.forEach((id, i) => {
    setTimeout(() => {
      document.getElementById(id).className = 'loading-step done';
      if (steps[i+1]) document.getElementById(steps[i+1]).className = 'loading-step active';
    }, i * 280);
  });
  setTimeout(() => {
    clearInterval(iv);
    bar.style.width = '100%';
  }, 1200);
}

function saveToHistory(result, imgSrc) {
  const entry = {
    id: Date.now(),
    name: result.display_name,
    className: result.class_name,
    confidence: result.confidence,
    confidence_pct: result.confidence_pct,
    is_healthy: result.is_healthy,

function updateAnalytics() {
  const h = state.history;
  const total    = h.length;
  const healthy  = h.filter(x => x.is_healthy).length;
  const diseased = total - healthy;
  const dRate    = total ? Math.round(diseased/total*100) + '%' : '0%';
  const avgConf  = total ? (h.reduce((s,x) => s+x.confidence,0)/total*100).toFixed(0)+'%' : '—';

  document.getElementById('aTotal').textContent   = total;
  document.getElementById('aDRate').textContent   = dRate;
  document.getElementById('aAvgConf').textContent = avgConf;
  document.getElementById('lgHealthy').textContent  = healthy;
  document.getElementById('lgDiseased').textContent = diseased;

  // Donut
  const circ = 251;
  const dOffset = total ? circ - (diseased/total) * circ : circ;
  const hOffset = total ? circ - (healthy/total) * circ : circ;
  document.getElementById('donutDiseased').style.strokeDashoffset = dOffset;
  const hEl = document.getElementById('donutHealthy');
  hEl.style.strokeDashoffset = hOffset;
  hEl.style.transform = `rotate(${-90 + (diseased/total||0)*360}deg)`;
  hEl.style.transformOrigin = '55px 55px';

  // Disease bar chart
  const counts = {};
  h.filter(x => !x.is_healthy).forEach(x => {
    counts[x.name] = (counts[x.name] || 0) + 1;
  });
  const sorted = Object.entries(counts).sort((a,b) => b[1]-a[1]).slice(0,6);
  const barChart = document.getElementById('diseaseBarChart');
  if (!sorted.length) {
    barChart.innerHTML = '<p style="color:var(--text-3);font-size:.85rem">No diseases detected yet</p>';
  } else {
    const max = sorted[0][1];
    barChart.innerHTML = sorted.map(([name,count]) => `
      <div class="bar-row">
        <div class="bar-label" title="${name}">${name}</div>
        <div class="bar-track">
          <div class="bar-fill" style="width:${(count/max*100)}%">
            <span>${count}</span>
          </div>
        </div>
      </div>`).join('');
  }

  // Severity chart
  const sevCounts = { None:0, Moderate:0, High:0, Critical:0 };
  h.forEach(x => { if (x.severity in sevCounts) sevCounts[x.severity]++; });
  const sevColors = { None:'var(--green-1)', Moderate:'var(--gold)', High:'var(--amber)', Critical:'var(--rust)' };
  const sevMax = Math.max(...Object.values(sevCounts), 1);
  const sevChart = document.getElementById('severityChart');
  sevChart.innerHTML = Object.entries(sevCounts).map(([sev,count]) => `
    <div class="bar-row">
      <div class="bar-label">${sev}</div>
      <div class="bar-track">
        <div class="bar-fill" style="width:${(count/sevMax*100)}%;background:${sevColors[sev]}">
          <span>${count}</span>
        </div>
      </div>
    </div>`).join('');
}

function loadSampleImage() {
  // Create a simple leaf-shaped sample canvas
  const canvas  = document.createElement('canvas');
  canvas.width  = 224; canvas.height = 224;
  const ctx     = canvas.getContext('2d');
  ctx.fillStyle = '#1a3d1e';
  ctx.fillRect(0, 0, 224, 224);
  ctx.fillStyle = '#2d6a2d';
  ctx.beginPath();
  ctx.ellipse(112, 112, 80, 100, 0, 0, Math.PI * 2);
  ctx.fill();
  ctx.strokeStyle = '#4aa369';
  ctx.lineWidth = 2;
  ctx.beginPath();
  ctx.moveTo(112, 20); ctx.lineTo(112, 200);

  ctx.fillStyle = '#8b4513';

    ctx.beginPath(); ctx.arc(x, y, 8, 0, Math.PI*2); ctx.fill();

    const file = new File([blob], 'sample-leaf.jpg', { type: 'image/jpeg' });
    handleFile(file);
    toast('info', '🌱 Sample image loaded — click Analyze!');

// ── Utilities ─────────────────────────────────────────────────
function timeAgo(iso) {
  const s = Math.floor((Date.now() - new Date(iso)) / 1000);
  if (s < 60)   return 'just now';
  if (s < 3600) return Math.floor(s/60) + 'm ago';
  if (s < 86400) return Math.floor(s/3600) + 'h ago';
  return Math.floor(s/86400) + 'd ago';
}

function formatDate(iso) {
  return new Date(iso).toLocaleDateString('en-IN', { day:'2-digit', month:'short', year:'2-digit', hour:'2-digit', minute:'2-digit' });
}
